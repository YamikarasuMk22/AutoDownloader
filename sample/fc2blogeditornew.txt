function AddTag(tag){
	var txt = document.getElementById('usertag').value;
	var re_space = /(\u3000|\u3001|\u002c)+| {2,}/g;

	if(txt){
		txt =  txt.replace(re_space, " ");
		var tags = txt.split(" ");
	}else{
		var tags = new Array();
	}

	var f = false;
	var tags2 = new Array();

	for(var i=0; i<tags.length; i++){
		if(tags[i] == tag){
			f = true;
		}else{
			tags2.push(tags[i]);
		}
	}

	if(!f){
		if(tags.length >= 1){
			if (tags.length >= 10){
				window.alert('入力できるタグは10個までです。');
			}else{
				txt = txt + ' ' + tag;
			}
		}else{
			txt = tag;
		}
	}else{
		txt = tags2.join(" ");
	}

	txt = txt.replace(re_space, " ");
	document.getElementById('usertag').value = txt;
}

function checkChargeSingle()
{
	var val = document.getElementById('entry_charge_single').value;
	if ( invalidChargeSingleCheck( document.getElementById('entry_charge_single_check') ) ) return true;

	var errmsg = '';

	val = val.replace(/(^\s+)|(\s+$)/g, "");

	if (val == '') {
		errmsg = '単体販売を行う場合は、価格を設定して下さい';
	}

	if ( errmsg == '' ) {
		if (val.match(/[^0-9]/g)) {
			errmsg = '単体販売価格は数値で入力してください';
		}
	}

	if ( errmsg == '' ) {
		num = eval(val);
		if (! (num >= 100 && num <= 100000) ) {
			errmsg = '単体販売価格は100～100,000円で入力してください';
		}
	}

<script type="text/javascript">
// ページの内容に修正がかかった時 別ベージに移動する際に確認ダイアログを出す
var execBeforeUnload = true;
var formSerializeArray = {};
var unload_body_fck = "";
var unload_extend_fck = "";
var unload_blomaga_fck = "";
// formのentry[*の値を配列で保持
function getFormSerializeArray(){
	var serializeArray = {};
	var len = 0;
	jQuery.each($('form').serializeArray(),function(){
		if (this.name.indexOf('entry[')!=0) return ;
		serializeArray[this.name] = this.value;
		len++;
	});
	serializeArray.length = len;
	return serializeArray;
}
// formの内容が変更されたかをチェック
function isFormEdit(){
	var isEdit = true;
	if (formSerializeArray.length) {
		var checkFormSerializeArray = getFormSerializeArray();
		for(var i in checkFormSerializeArray){
			switch(i){
				case 'entry[eno]': break;
				case 'entry[body]': case 'entry[extend]': case 'entry[blomaga_intro]':
					var checkText = checkFormSerializeArray[i].replace(/(?:<p>(?:&nbsp;|\s*)<\/p>|<br(?:\s?\/)?>)|(?:&nbsp;)|(?:&#160;)*$/i, '');
					isEdit = isEdit && (checkFormSerializeArray[i] == formSerializeArray[i] || checkText == formSerializeArray[i]);
					break;
				default:
					isEdit = isEdit && checkFormSerializeArray[i] == formSerializeArray[i];
					break;
			}
		}
	}
	return !isEdit;
}
// FCKの確認ダイアログ用初期値設定
function setUnloadFckBody(){
	if (!Editor._status.isFck['body'] || !Editor._instances.fckeditor['body'] || !Editor._instances.fckeditor['body'].GetXHTML) {
		setTimeout("setUnloadFckBody()", 1000);
		return ;
	}
	unload_body_fck = Editor._instances.fckeditor['body'].GetXHTML();
}
function setUnloadFckExtend(){
	if (!Editor._status.isFck['extend'] || !Editor._instances.fckeditor['extend'] || !Editor._instances.fckeditor['extend'].GetXHTML) {
		setTimeout("setUnloadFckExtend()", 1000);
		return ;
	}
	unload_extend_fck = Editor._instances.fckeditor['extend'].GetXHTML();
}
function setUnloadFckBlomaga(){
	if (!Editor._status.isFck['blomaga_intro'] || !Editor._instances.fckeditor['blomaga_intro'] || !Editor._instances.fckeditor['blomaga_intro'].GetXHTML) {
		setTimeout("setUnloadFckExtend()", 1000);
		return ;
	}
	unload_blomaga_fck = Editor._instances.fckeditor['blomaga_intro'].GetXHTML();
}
// ゴミ文字対応用
function unloadFckComp(u_html, n_html){
	return u_html.replace(/(?:<p>(?:&nbsp;|\s*)<\/p>|<br(?:\s?\/)?>)|(?:&nbsp;)|(?:&#160;)*$/i, '') ==
		n_html.replace(/(?:<p>(?:&nbsp;|\s*)<\/p>|<br(?:\s?\/)?>)|(?:&nbsp;)|(?:&#160;)*$/i, '');
}
// IEのFlashでbeforeunloadが呼ばれる対策用
function isMatchFlashPosition(x, y, swf){
	if( swf.css('display') == 'none' ){
		// 非表示の場合は計算に入れない
		return false;
	}
	var top = swf.offset().top;
	var left = swf.offset().left;
	var height = swf.height();
	var width = swf.width();
	if(top <= y && y <= top + height){
		if(left <= x && x <= left + width){
			return true;
		}
	}
	return false;
}
$(function(){
	if(!$('#body').length){
		// 投稿と完了が同一テンプレートなので分岐処理代わり
		return ;
	}
	// カテゴリーテキストボックスEnter時の処理
	$('input[name=name]').on('keypress', function(e){
		if (e.keyCode==13) {
			execBeforeUnload = false;
			Editor.commands.submit('category');
			return false;
		}
	});

	// Unloadイベント系の処理
	formSerializeArray = getFormSerializeArray();
	setUnloadFckBody();
	setUnloadFckExtend();
	$(window).on('beforeunload', function(e) {
		// 入力の変更があったかどうかのチェック
		var isData = !isFormEdit();
		if (Editor._status.isFck['body']) {
			isData = isData && unloadFckComp(unload_body_fck, Editor._instances.fckeditor['body'].GetXHTML());
		}
		if (Editor._status.isFck['extend']) {
			isData = isData && unloadFckComp(unload_extend_fck, Editor._instances.fckeditor['extend'].GetXHTML());
		}
		if (Editor._status.isFck['blomaga_intro']) {
			isData = isData && unloadFckComp(unload_blomaga_fck, Editor._instances.fckeditor['blomaga_intro'].GetXHTML());
		}

		// IEの場合のみFlashの位置に座標があった場合スルー(Flashのイベントでonbeforeunloadが呼ばれる為
		var isMSIE = /*@cc_on!@*/false;
		if (isMSIE) {
			if (e && e.currentTarget && e.currentTarget.event && e.currentTarget.event.clientX) {
				var x = e.currentTarget.event.clientX;
				var y = e.currentTarget.event.clientY;
				x += document.documentElement.scrollLeft || document.body.scrollLeft;
				y += document.documentElement.scrollTop || document.body.scrollTop;
				if (isMatchFlashPosition(x, y, $('#swf')) || isMatchFlashPosition(x, y, $('#swf2'))) {
					// どちらかにマッチした場合 onbeforeunload で確認ダイアログは表示しない
					isData = true;
				}
			}
		}

		// 変更があった場合 移動前に確認ダイアログを表示
		if( isData == false ){
			if (execBeforeUnload) {
				var unload_message = '保存が完了しておりません。保存していないデータが失われますがよろしいですか？';
				if(unload_message != ''){
					return unload_message;
				}
			}
		}
	});
});
